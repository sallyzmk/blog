```
cd home/
cd human/
cd airflow/

# 모든 파일 삭제
rm -rf *

# 설치
sudo apt-get update

sudo apt install python3-pip

sudo pip3 install virtualenv

# 가상환경 설치
virtualenv venv

# 리눅스는 bin/ 윈도우는 Script
source venv/bin/activate

# venv 가상환경 들어왔는지 확인
ls

# 경로 확인 및 복사 /home/human/airflow
pwd

# <실무 예제로 배우는 데이터 공학> 25p
pip3 install 'apache-airflow[postgres, slack, celery]'

# 이상한 코드 주루룩 뜸
vi ~/.bashrc
```
